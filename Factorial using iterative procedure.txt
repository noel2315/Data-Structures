26.Find the factorial of a number using iterative procedure
Input : 3
Output: 6


ans:
#include<stdio.h>


int main() {
    int n, f = 1;
    printf("Enter a number: ");
    scanf("%d", &n);
    for(int i = 1; i <= n; i++) f *= i;
    printf("Factorial of %d is %d\n", n, f);
    return 0;
}
